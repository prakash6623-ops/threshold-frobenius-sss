# Automorphism-Compatible Shamir Secret Sharing over F_{p^2}
# Example over F_49 for threshold isogeny cryptography
# Authors: Tikaram, Adhikari, KC, Kumar, Pandey

p = 7
prime_field = GF(p)
R.<omega> = PolynomialRing(prime_field)
quadratic_field.<w> = prime_field.extension(omega^2 + 1)

print("Field:", quadratic_field)
print("Characteristic:", quadratic_field.characteristic())
print("Size:", quadratic_field.cardinality())

t = 3
n = 5
eval_points = [prime_field(i) for i in range(1, n+1)]
print("\nEvaluation points:", eval_points)

secret = 1 + 2*w
coeffs = [secret, w, 1 + w]
R_poly.<x> = quadratic_field[]
f = sum(coeffs[i] * x^i for i in range(3))
print("\nSecret S =", secret)
print("Polynomial f(x) =", f)

shares = [(alpha, f(alpha)) for alpha in eval_points]
print("\nShares:")
for alpha, s in shares:
    print(f"  {alpha} -> {s}")

def interpolate(points, x0=0):
    xs = [p[0] for p in points]
    ys = [p[1] for p in points]
    k = len(xs)
    result = quadratic_field(0)
    for j in range(k):
        term = ys[j]
        for m in range(k):
            if m == j:
                continue
            term *= (x0 - xs[m]) / (xs[j] - xs[m])
        result += term
    return result

test = shares[:t]
reconstructed = interpolate(test)
print(f"\nReconstructed S from {t} shares: {reconstructed}")
assert reconstructed == secret
print("✓ Correct")

def frobenius_map(a):
    return a^p

for alpha in eval_points:
    assert frobenius_map(alpha) == alpha
print("\n✓ Frobenius fixes evaluation points")

transformed = [(alpha, frobenius_map(s)) for (alpha, s) in shares]
reconstructed_psi = interpolate(transformed[:t])
print(f"Reconstructed ψ(S): {reconstructed_psi}")
print(f"Direct ψ(S): {frobenius_map(secret)}")
assert reconstructed_psi == frobenius_map(secret)
print("✓ Non-interactive transformation works")
